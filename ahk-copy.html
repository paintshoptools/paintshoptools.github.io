#Requires AutoHotkey v2.0
#SingleInstance Force

; Hotkey: Ctrl+Shift+J
^+j::
{
	try {
		; Get JSON string from clipboard
		jsonText := A_Clipboard
		
		if (jsonText = "")
		{
			MsgBox "Clipboard is empty!"
			return
		}
		
		; Parse the JSON string
		data := JSON.Parse(jsonText)

		; Iterate through objects in the array
		for obj in data
		{
			SendText obj.name
			Send "{Tab}"
			
			; Send the "values" elements separated by tabs
			for val in obj.values
			{
				SendText val
				Send "{Tab}"
			}
		}
	}
	catch e {
		MsgBox "Error: " e.Message
	}
}
